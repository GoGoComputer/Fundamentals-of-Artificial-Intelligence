# 1장 종합 실습: 계산기 만들기

- 원본 파일: `01-파이썬-기초/실습/05_계산기.py`
- 동기화 방식: 강의 원본에서 자동 생성

아래는 생략 없이 전체 코드입니다.

```python
"""1장 종합 실습: 계산기 만들기

지금까지 배운 변수, 조건문, 반복문, 함수, 입력 처리를 모두 합친 작은 프로그램입니다.
코드가 조금 길어 보여도 역할이 뚜렷하게 나뉘어 있으니 주석을 따라가며 읽어 보세요.

Colab에서는 input()이 잘 동작하지 않을 수 있어요.
그럴 땐 본인 컴퓨터에서:
    python 05_계산기.py
이렇게 실행하시거나, Colab에서는 아래의 calculate() 함수만 가져다 쓰셔도 됩니다.
"""


# 핵심 계산 함수를 소개하는 구간이라는 점을 구분선 대신 주석으로 먼저 알려 줍니다.

# calculate 함수는 숫자 두 개와 연산자 하나를 받아 실제 계산을 담당합니다.
def calculate(a, b, op):
    """두 숫자와 연산자를 받아 결과를 돌려준다.

    지원 연산: +, -, *, /, **, %
    잘못된 입력이거나 0으로 나누기는 None을 돌려준다.
    """
    # 더하기 기호가 들어오면 두 수를 더한 값을 반환합니다.
    if op == "+":
        return a + b
    # 빼기 기호가 들어오면 왼쪽에서 오른쪽을 뺀 값을 반환합니다.
    elif op == "-":
        return a - b
    # 곱하기 기호가 들어오면 두 수의 곱을 반환합니다.
    elif op == "*":
        return a * b
    # 나누기 기호가 들어오면 먼저 0으로 나누는 상황인지 확인합니다.
    elif op == "/":
        # 분모가 0이면 계산할 수 없으므로 None을 반환합니다.
        if b == 0:
            return None
        # 정상적인 경우에는 나눗셈 결과를 반환합니다.
        return a / b
    # 거듭제곱 기호가 들어오면 a의 b제곱을 계산합니다.
    elif op == "**":
        return a ** b
    # 나머지 연산도 0으로 나누는지 먼저 확인해야 합니다.
    elif op == "%":
        # 분모가 0이면 역시 계산 불가이므로 None을 반환합니다.
        if b == 0:
            return None
        # 정상일 때는 나머지 값을 반환합니다.
        return a % b
    else:
        # 지원하지 않는 연산자가 들어오면 None으로 실패를 알려 줍니다.
        return None


# parse_number 함수는 사용자가 입력한 글자를 실제 숫자로 바꾸는 역할을 맡습니다.
def parse_number(text):
    """문자열을 숫자로 변환. 실패하면 None."""
    # 변환이 성공할 수도 있고 실패할 수도 있으므로 try로 시도합니다.
    try:
        # float로 바꾸면 정수와 실수를 모두 유연하게 받을 수 있습니다.
        return float(text)
    except ValueError:
        # 숫자로 바꿀 수 없는 글자가 들어오면 None을 반환해 실패를 표시합니다.
        return None


# 메인 루프는 사용자와 계속 대화하면서 계산을 반복하는 부분입니다.

# main 함수는 실제 계산기 프로그램의 흐름 전체를 관리합니다.
def main():
    # 시작 화면의 윗줄 테두리를 출력합니다.
    print("=" * 40)
    # 프로그램 이름을 가운데 느낌으로 보여 줍니다.
    print("  파이썬 계산기")
    # q를 입력하면 종료된다는 사용법도 함께 안내합니다.
    print("  (q를 입력하면 종료)")
    # 시작 화면 아래쪽 테두리도 출력합니다.
    print("=" * 40)

    # 계산 기록을 저장할 빈 리스트를 먼저 준비합니다.
    history = []

    # 사용자가 종료할 때까지 계속 입력을 받기 위해 무한 반복을 시작합니다.
    while True:
        # 보기 좋게 한 줄을 비운 뒤 새 계산을 시작합니다.
        print()
        # 첫 번째 숫자를 문자열로 입력받고 양쪽 공백은 제거합니다.
        a_text = input("첫 번째 숫자: ").strip()
        # 사용자가 q를 입력했다면 프로그램을 종료합니다.
        if a_text.lower() == "q":
            break

        # 두 번째 숫자도 같은 방식으로 입력받습니다.
        b_text = input("두 번째 숫자: ").strip()
        # 여기서도 q를 입력하면 종료할 수 있게 합니다.
        if b_text.lower() == "q":
            break

        # 연산자를 입력받아 어떤 계산을 할지 정합니다.
        op = input("연산 (+, -, *, /, **, %): ").strip()
        # 연산자 입력 단계에서도 q를 입력하면 종료할 수 있습니다.
        if op.lower() == "q":
            break

        # 입력받은 첫 번째 문자열을 실제 숫자로 바꿉니다.
        a = parse_number(a_text)
        # 두 번째 문자열도 숫자로 바꿉니다.
        b = parse_number(b_text)

        # 둘 중 하나라도 숫자로 변환하지 못했다면 사용자에게 다시 입력을 요청합니다.
        if a is None or b is None:
            print("→ 숫자가 잘못됐어요. 다시 시도하세요.")
            continue

        # 숫자 변환이 끝났으면 이제 실제 계산 함수를 호출합니다.
        result = calculate(a, b, op)
        # 결과가 None이면 계산 불가 상황이므로 메시지를 출력하고 다음 반복으로 넘어갑니다.
        if result is None:
            print("→ 계산할 수 없습니다 (잘못된 연산이거나 0으로 나누기)")
            continue

        # 사람이 읽기 쉬운 계산 기록 문자열을 만듭니다.
        record = f"{a} {op} {b} = {result}"
        # 방금 계산한 결과를 화면에 보여 줍니다.
        print(f"→ {record}")
        # 같은 기록을 history 리스트에도 저장해 종료할 때 다시 보여 줄 수 있게 합니다.
        history.append(record)

    # 종료할 때 기록이 하나라도 있으면 요약 화면을 보여 줍니다.
    if history:
        # 기록 섹션의 윗부분 테두리를 출력합니다.
        print("\n" + "=" * 40)
        # 지금까지의 계산 목록이라는 제목을 출력합니다.
        print("  지금까지의 계산")
        # 아래쪽 테두리를 출력합니다.
        print("=" * 40)
        # enumerate를 이용해 번호와 기록을 함께 출력합니다.
        for i, record in enumerate(history, start=1):
            print(f"  {i}. {record}")

    # 프로그램이 완전히 끝난다는 인사말을 마지막에 출력합니다.
    print("\n계산기를 종료합니다. 안녕히가세요!")


# 실행 전에 계산 함수가 제대로 동작하는지 확인하는 테스트 구간입니다.

# test_calculate 함수는 여러 입력으로 calculate를 점검하는 작은 안전망입니다.
def test_calculate():
    """calculate 함수가 잘 동작하는지 검증.

    이런 식으로 함수마다 테스트를 적어 두면, 코드를 고칠 때
    뭐가 깨졌는지 바로 알 수 있어요. 이게 좋은 코딩 습관입니다.
    """
    # 더하기가 정상 동작하는지 확인합니다.
    assert calculate(10, 3, "+") == 13
    # 빼기도 기대한 결과를 내는지 확인합니다.
    assert calculate(10, 3, "-") == 7
    # 곱하기도 같은 방식으로 검증합니다.
    assert calculate(10, 3, "*") == 30
    # 나누기 결과가 실수로 잘 계산되는지 확인합니다.
    assert calculate(10, 4, "/") == 2.5
    # 0으로 나누기는 None 처리되는지 검사합니다.
    assert calculate(10, 0, "/") is None
    # 거듭제곱 연산도 올바른지 확인합니다.
    assert calculate(2, 10, "**") == 1024
    # 나머지 연산도 기대값과 비교합니다.
    assert calculate(10, 3, "%") == 1
    # 지원하지 않는 연산자는 None으로 처리되는지 확인합니다.
    assert calculate(10, 3, "?") is None
    # 모든 검사가 통과했다는 메시지를 출력합니다.
    print("[테스트 통과] calculate 함수 OK")


# 이 아래 구역은 파일을 직접 실행했을 때만 동작하는 진입점입니다.

# 현재 파일이 직접 실행된 상황인지 확인합니다.
if __name__ == "__main__":
    # 본 프로그램에 들어가기 전에 테스트부터 먼저 실행합니다.
    test_calculate()
    # 테스트 결과와 본 프로그램 화면이 붙지 않도록 한 줄 비웁니다.
    print()

    # 테스트가 통과했으면 이제 계산기 프로그램 본체를 실행합니다.
    main()
```
