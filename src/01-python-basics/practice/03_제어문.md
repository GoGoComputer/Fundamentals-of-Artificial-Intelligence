# 1장 5절 실습: if, for, while

- 원본 파일: `01-파이썬-기초/실습/03_제어문.py`
- 동기화 방식: 강의 원본에서 자동 생성

아래는 생략 없이 전체 코드입니다.

```python
"""1장 5절 실습: if, for, while

제어문은 조건에 따라 다른 길로 가거나 같은 일을 여러 번 반복하게 만드는 도구입니다.
한 줄씩 결과를 확인하면서 흐름이 어디서 갈라지는지 꼭 눈으로 따라가 보세요.
"""

# 첫 번째 주제인 조건문 섹션을 구분선으로 시작합니다.
print("=" * 50)
# 지금부터 if 문법을 본다는 제목을 출력합니다.
print("1. 조건문")
# 제목 아래에도 구분선을 넣어 화면을 정리합니다.
print("=" * 50)

# 나이 값을 하나 준비해서 조건 비교 예제를 시작합니다.
age = 25
# age가 19 이상일 때만 아래 들여쓴 코드가 실행됩니다.
if age >= 19:
    # 조건이 참이므로 성인이라는 문장이 출력됩니다.
    print("성인입니다")

# 점수 값을 바꾸어 합격과 불합격이 갈리는 경우를 실습합니다.
score = 75
# 점수가 60 이상이면 합격 쪽 블록이 실행됩니다.
if score >= 60:
    # 이번 예시에서는 75점이므로 이 줄이 실행됩니다.
    print("합격")
else:
    # 조건이 거짓일 때는 이 줄이 대신 실행됩니다.
    print("불합격")

# 여러 조건을 순서대로 검사하는 학점 예제를 준비합니다.
score = 85
# 가장 높은 조건부터 검사해야 뒤쪽 조건이 먼저 잡아먹지 않습니다.
if score >= 90:
    # 90점 이상이면 A를 부여합니다.
    grade = "A"
elif score >= 80:
    # 80점 이상 90점 미만이면 B를 부여합니다.
    grade = "B"
elif score >= 70:
    # 70점 이상 80점 미만이면 C를 부여합니다.
    grade = "C"
else:
    # 위 조건에 모두 해당하지 않으면 F로 처리합니다.
    grade = "F"
# 최종 학점을 출력해 분기 결과를 확인합니다.
print(f"학점: {grade}")

# 빈 리스트는 거짓처럼 취급되므로 not을 붙이면 참이 됩니다.
items = []
# 그래서 빈 리스트일 때만 아래 문장이 실행됩니다.
if not items:
    print("리스트가 비어있어요")

# 빈 문자열도 같은 규칙을 따르는지 확인하기 위해 값을 비워 둡니다.
text = ""
# 문자열이 비어 있으면 역시 not 조건이 참이 됩니다.
if not text:
    print("문자열이 비어있어요")


# 이제 반복문 섹션으로 넘어간다는 사실을 구분선으로 보여 줍니다.
print("\n" + "=" * 50)
# 가장 자주 쓰는 반복문인 for 제목을 출력합니다.
print("2. for문")
# 동일한 형식의 구분선을 유지합니다.
print("=" * 50)

# 과일 리스트를 하나 준비해서 원소를 하나씩 순회해 보겠습니다.
fruits = ["사과", "바나나", "딸기"]
# for는 리스트 안의 값을 앞에서부터 차례대로 fruit 변수에 넣어 줍니다.
for fruit in fruits:
    # 현재 꺼낸 값을 화면에 출력하면 반복이 어떻게 도는지 눈에 보입니다.
    print(f"  - {fruit}")

# range는 연속된 숫자를 만들어 반복문에 넘겨 주는 도구입니다.
print("\n0부터 4까지:")
# range(5)는 0, 1, 2, 3, 4를 차례대로 만들어 냅니다.
for i in range(5):
    # end를 공백으로 지정해 줄바꿈 대신 옆으로 이어서 출력합니다.
    print(i, end=" ")
# 숫자 출력이 끝난 뒤에는 줄을 한 번 바꿔 보기 좋게 정리합니다.
print()

# 시작과 끝을 주면 원하는 구간의 숫자만 반복할 수 있습니다.
print("\n2부터 7 미만, 1씩:")
# 이 반복은 2, 3, 4, 5, 6까지만 생성합니다.
for i in range(2, 7):
    # 만들어진 숫자를 한 칸씩 띄워서 출력합니다.
    print(i, end=" ")
# 다음 예제를 위해 줄을 바꿉니다.
print()

# step을 2로 주면 두 칸씩 건너뛰며 숫자를 생성할 수 있습니다.
print("\n0부터 10 미만, 2씩:")
# 그래서 짝수만 출력하는 반복이 자연스럽게 만들어집니다.
for i in range(0, 10, 2):
    # 현재 숫자를 한 줄에 이어서 출력합니다.
    print(i, end=" ")
# 세 번째 range 예제도 여기서 마무리합니다.
print()

# enumerate를 쓰면 값뿐 아니라 현재 순번도 함께 받을 수 있습니다.
print("\nenumerate:")
# i에는 인덱스가 들어가고 fruit에는 실제 값이 들어갑니다.
for i, fruit in enumerate(fruits):
    # 인덱스와 값을 같이 출력하면 enumerate의 장점이 바로 드러납니다.
    print(f"  {i}번째: {fruit}")

# 딕셔너리 순회는 설정값처럼 이름표가 있는 데이터를 볼 때 자주 씁니다.
print("\n딕셔너리 순회:")
# 학습 설정처럼 보이는 딕셔너리를 하나 만들어 둡니다.
config = {
    "learning_rate": 0.001,
    "epochs": 10,
    "batch_size": 32,
}
# items를 쓰면 키와 값을 함께 받아 보기 좋게 출력할 수 있습니다.
for key, value in config.items():
    print(f"  {key}: {value}")

# 누적 합계는 반복문에서 가장 자주 만나는 고전 패턴입니다.
nums = [10, 20, 30, 40, 50]
# 합계를 담을 변수를 먼저 0으로 초기화합니다.
total = 0
# 숫자를 하나씩 꺼내면서 total에 계속 더해 나갑니다.
for n in nums:
    total += n
# 반복이 끝난 뒤에는 모든 값이 합쳐진 결과가 남아 있습니다.
print(f"\n누적 합: {total}")

# 같은 결과를 내는 내장 함수 sum도 있다는 점을 비교용으로 함께 보여 줍니다.
print(f"sum() 한 번에: {sum(nums)}")


# 이제 조건이 만족되는 동안 계속 도는 while 문으로 넘어갑니다.
print("\n" + "=" * 50)
# 현재 섹션 제목을 출력합니다.
print("3. while문")
# 같은 형식의 구분선을 반복합니다.
print("=" * 50)

# 카운터 변수를 0으로 두고 반복 시작점을 만듭니다.
count = 0
# count가 5보다 작은 동안은 반복이 계속됩니다.
while count < 5:
    # 현재 count 값을 출력하면 while의 진행 상태가 보입니다.
    print(count, end=" ")
    # count를 1씩 늘려 주지 않으면 반복이 끝나지 않으므로 꼭 증가시킵니다.
    count += 1
# 반복 종료 뒤에는 줄을 바꿔 다음 출력이 붙지 않게 합니다.
print()

# 조금 더 흥미로운 반복을 위해 시작값을 100으로 둡니다.
n = 100
# 몇 번 반복했는지 세기 위한 steps도 0으로 시작합니다.
steps = 0
# n이 1보다 큰 동안에는 규칙을 계속 적용합니다.
while n > 1:
    # n이 짝수면 2로 나누는 규칙을 적용합니다.
    if n % 2 == 0:
        n = n // 2
    else:
        # n이 홀수면 3을 곱하고 1을 더하는 규칙을 적용합니다.
        n = 3 * n + 1
    # 한 번 규칙을 적용할 때마다 단계 수를 1 증가시킵니다.
    steps += 1
# 최종적으로 몇 단계가 걸렸는지 출력해 반복 계산이 끝났음을 확인합니다.
print(f"100부터 시작해서 1이 되기까지 {steps}단계")


# 반복을 중간에 끊거나 건너뛰는 제어 키워드 섹션을 시작합니다.
print("\n" + "=" * 50)
# 현재 다룰 주제를 제목으로 출력합니다.
print("4. break와 continue")
# 시각적인 구분선을 한 번 더 출력합니다.
print("=" * 50)

# break는 반복문을 즉시 종료시킨다는 점을 실험합니다.
print("5에서 멈춤:")
# 0부터 19까지 돌도록 반복을 시작합니다.
for n in range(20):
    # n이 5가 되면 break가 실행되어 반복이 완전히 끝납니다.
    if n == 5:
        break
    # break 전에 도달한 값들만 출력됩니다.
    print(n, end=" ")
# 다음 출력과 붙지 않도록 줄을 정리합니다.
print()

# continue는 이번 한 번만 건너뛰고 다음 반복으로 넘어갈 때 사용합니다.
print("\n홀수만:")
# 0부터 9까지 돌면서 짝수는 건너뛰어 보겠습니다.
for n in range(10):
    # 짝수일 때는 continue로 아래 출력 줄을 건너뜁니다.
    if n % 2 == 0:
        continue
    # 그래서 결과적으로 홀수만 출력됩니다.
    print(n, end=" ")
# 출력 형식을 정리하기 위해 마지막 줄바꿈을 추가합니다.
print()


# 지금까지 배운 제어문을 한 번에 합치는 종합 예제로 들어갑니다.
print("\n" + "=" * 50)
# 학생 점수 처리라는 실제 느낌의 제목을 출력합니다.
print("5. 종합 예제: 학생 점수")
# 동일한 형식으로 구분선을 출력합니다.
print("=" * 50)

# 학생 이름과 점수를 딕셔너리로 정리해 두면 반복문으로 다루기 좋습니다.
students = {
    "철수": 85,
    "영희": 92,
    "민수": 78,
    "수진": 95,
    "지호": 67,
}

# 먼저 전체 학생 목록을 한 번 출력해 데이터가 어떤 모양인지 확인합니다.
print("[전체 학생]")
# items를 이용해 이름과 점수를 함께 꺼내 순회합니다.
for name, score in students.items():
    print(f"  {name}: {score}점")

# values만 꺼내 합계를 구하고 학생 수로 나눠 평균을 계산합니다.
avg = sum(students.values()) / len(students)
# 소수점 한 자리까지만 보이도록 형식을 지정해 출력합니다.
print(f"\n평균: {avg:.1f}점")

# 이번에는 80점 이상 학생만 걸러내는 조건문을 반복문 안에 넣어 봅니다.
print("\n[합격자 (80점 이상)]")
# 다시 모든 학생을 순회하면서 조건을 검사합니다.
for name, score in students.items():
    # 80점 이상이면 합격자 목록에 출력합니다.
    if score >= 80:
        print(f"  {name}: {score}점")

# 최고점을 찾기 위해 현재 최고 이름과 점수를 초기값으로 준비합니다.
best_name, best_score = "", 0
# 다시 전체 학생을 돌면서 더 높은 점수가 나오면 기록을 갱신합니다.
for name, score in students.items():
    if score > best_score:
        best_name, best_score = name, score
# 반복이 끝나면 최고점 학생 정보가 남아 있으므로 출력합니다.
print(f"\n최고점: {best_name} ({best_score}점)")


# 마지막으로 짧지만 자주 쓰이는 리스트 컴프리헨션을 맛보기로 봅니다.
print("\n" + "=" * 50)
# 보너스 섹션 제목을 출력합니다.
print("보너스: 리스트 컴프리헨션")
# 같은 형식의 구분선을 유지합니다.
print("=" * 50)

# 먼저 익숙한 for문으로 짝수 리스트를 만들어 봅니다.
evens = []
# 0부터 9까지 순회합니다.
for n in range(10):
    # 짝수일 때만 리스트에 추가합니다.
    if n % 2 == 0:
        evens.append(n)
# 완성된 리스트를 출력합니다.
print("for문:", evens)

# 같은 일을 한 줄로 적은 버전이 리스트 컴프리헨션입니다.
evens = [n for n in range(10) if n % 2 == 0]
# 결과가 위와 같다는 점을 직접 비교합니다.
print("컴프리헨션:", evens)

# 각 숫자의 제곱을 모은 새 리스트도 같은 문법으로 만들 수 있습니다.
squares = [n * n for n in range(10)]
# 만들어진 제곱 리스트를 출력해 문법 감각을 잡습니다.
print("제곱:", squares)


# 마무리 문장을 출력해 제어문 실습이 끝났음을 알려 줍니다.
print("\n실습 끝! 제어문은 머신러닝의 모든 곳에서 만납니다.")
```
