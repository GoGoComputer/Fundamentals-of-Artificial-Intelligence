# 1장 4절 실습: 리스트, 튜플, 셋, 딕셔너리

- 원본 파일: `01-파이썬-기초/실습/02_자료구조.py`
- 동기화 방식: 강의 원본에서 자동 생성

아래는 생략 없이 전체 코드입니다.

```python
"""1장 4절 실습: 리스트, 튜플, 셋, 딕셔너리

각 섹션을 차례로 실행하시면서 결과를 확인하세요.
같은 코드를 조금씩 바꿔 보면서 손으로 감각을 만드는 것이 가장 중요합니다.
"""

# 첫 번째 자료구조인 리스트를 시작한다는 사실을 구분선으로 보여 줍니다.
print("=" * 50)
# 지금 보고 있는 자료구조의 이름을 제목으로 출력합니다.
print("1. 리스트")
# 섹션 제목 아래에도 선을 하나 더 깔아 화면 구성을 안정적으로 만듭니다.
print("=" * 50)

# 과일 세 개를 순서대로 담은 리스트를 만들어 가장 기본적인 형태를 준비합니다.
fruits = ["사과", "바나나", "딸기"]
# 리스트를 통째로 출력하면 대괄호 안에 원소가 순서대로 보입니다.
print(fruits)

# 첫 번째 값은 0번 인덱스로 꺼낸다는 점을 직접 확인합니다.
print("첫 번째:", fruits[0])
# 두 번째 값은 1번 인덱스로 꺼내며 번호가 0부터 시작함을 익힙니다.
print("두 번째:", fruits[1])
# 마지막 값은 -1을 쓰면 뒤에서부터 접근할 수 있어 편리합니다.
print("마지막:", fruits[-1])

# 숫자 리스트를 새로 만들어 슬라이싱 예제를 따로 실습하겠습니다.
nums = [10, 20, 30, 40, 50]
# 1번부터 3번 인덱스까지 잘라 오면 중간 구간만 뽑을 수 있습니다.
print(nums[1:4])
# 시작 위치를 비우면 맨 앞부터 자른다는 점도 함께 확인합니다.
print(nums[:3])
# 끝 위치를 비우면 지정한 지점부터 끝까지 가져옵니다.
print(nums[2:])
# step을 -1로 주면 리스트가 거꾸로 뒤집혀서 보입니다.
print(nums[::-1])

# 리스트는 수정 가능한 자료구조이므로 두 번째 값을 포도로 바꿔 봅니다.
fruits[1] = "포도"
# 수정 후 리스트를 다시 출력해 실제로 값이 바뀌었는지 확인합니다.
print("수정 후:", fruits)

# append는 맨 뒤에 새 값을 하나 추가할 때 가장 자주 쓰는 메서드입니다.
fruits.append("수박")
# 값이 맨 끝에 붙었는지 바로 출력해서 눈으로 확인합니다.
print("추가 후:", fruits)

# remove는 값 자체를 찾아서 삭제하므로 여기서는 사과가 사라집니다.
fruits.remove("사과")
# 삭제 후 결과를 보면 리스트 길이와 순서가 함께 바뀐다는 점을 볼 수 있습니다.
print("삭제 후:", fruits)

# 정렬 예제를 위해 섞여 있는 숫자 리스트를 새로 준비합니다.
nums = [3, 1, 4, 1, 5, 9, 2, 6]
# sort는 원본 리스트 자체를 오름차순으로 바꿉니다.
nums.sort()
# 정렬 결과를 보면 숫자가 작은 것부터 큰 것 순으로 배열됩니다.
print("정렬:", nums)

# reverse=True를 주면 이번에는 내림차순 정렬이 됩니다.
nums.sort(reverse=True)
# 큰 수부터 작은 수 순서로 바뀐 결과를 출력해 확인합니다.
print("역순 정렬:", nums)

# len은 리스트 안에 원소가 몇 개인지 알려 줍니다.
print("개수:", len(nums))
# sum은 숫자 리스트의 합계를 빠르게 구할 때 매우 편합니다.
print("합:", sum(nums))
# max는 가장 큰 값을 찾는 기본 함수입니다.
print("최대:", max(nums))
# min은 가장 작은 값을 찾는 기본 함수입니다.
print("최소:", min(nums))

# 리스트 안에 리스트를 넣어 2차원 구조를 만든 뒤 표처럼 다뤄 보겠습니다.
matrix = [
    [1, 2, 3],
    [4, 5, 6],
    [7, 8, 9],
]
# 첫 번째 인덱스로 행을 고르고 두 번째 인덱스로 열을 골라 원하는 값을 꺼냅니다.
print("2x2 위치:", matrix[1][1])

# 다음은 값을 바꿀 수 없는 튜플 섹션이라는 점을 화면에 표시합니다.
print("\n" + "=" * 50)
# 튜플 제목을 출력해 리스트와 구분합니다.
print("2. 튜플")
# 같은 형식의 구분선을 유지해 학습 리듬을 일정하게 만듭니다.
print("=" * 50)

# 좌표처럼 순서는 있지만 바뀌면 곤란한 값을 튜플로 표현합니다.
point = (3, 5)
# 색상 이름도 튜플에 담아 인덱싱은 가능하다는 점을 보여 줍니다.
colors = ("빨강", "초록", "파랑")
# 튜플 전체를 출력하면 소괄호 형태로 보입니다.
print(point)
# 튜플도 리스트처럼 인덱스로 값 하나를 꺼낼 수 있습니다.
print(colors[1])

# 언패킹을 이용하면 튜플의 각 값을 변수 두 개에 한 번에 나눠 담을 수 있습니다.
x, y = point
# 그래서 x와 y가 각각 어떤 값을 받았는지 바로 출력해 봅니다.
print(f"x={x}, y={y}")

# 여러 값을 돌려주는 함수 예제를 만들기 위해 최소값과 최대값 함수를 정의합니다.
def get_min_max(numbers):
    # min과 max를 같이 돌려주면 자동으로 튜플 형태가 됩니다.
    return min(numbers), max(numbers)

# 함수 결과를 두 변수에 바로 받아 언패킹이 실전에서 어떻게 쓰이는지 보여 줍니다.
mn, mx = get_min_max([3, 7, 1, 9, 4])
# 최소값과 최대값이 각각 잘 들어갔는지 출력합니다.
print(f"min={mn}, max={mx}")

# 셋은 중복이 없고 순서가 없다는 점을 실험하는 구역입니다.
print("\n" + "=" * 50)
# 현재 섹션 제목을 출력합니다.
print("3. 셋")
# 시각적인 구분선을 한 번 더 출력합니다.
print("=" * 50)

# 중복이 있는 리스트를 준비해서 set이 어떤 일을 하는지 보겠습니다.
nums = [1, 2, 2, 3, 3, 3, 4, 5]
# 리스트를 set으로 바꾸면 같은 값이 자동으로 하나만 남습니다.
unique = set(nums)
# 중복 제거 결과를 출력해 순서가 보장되지 않는다는 점도 함께 확인합니다.
print(unique)

# 집합 연산을 위해 첫 번째 셋을 만듭니다.
A = {1, 2, 3, 4}
# 두 번째 셋을 만들어 비교 대상도 준비합니다.
B = {3, 4, 5, 6}

# | 연산자는 두 집합의 모든 원소를 합친 합집합을 의미합니다.
print("합집합:", A | B)
# & 연산자는 두 집합에 공통으로 들어 있는 값만 남기는 교집합입니다.
print("교집합:", A & B)
# - 연산자는 왼쪽 집합에만 있고 오른쪽에는 없는 값만 남기는 차집합입니다.
print("차집합:", A - B)

# 셋에도 새 값을 추가할 수 있다는 점을 보여 주기 위해 과일 셋을 만듭니다.
fruits = {"사과", "바나나"}
# add 메서드로 딸기를 추가합니다.
fruits.add("딸기")
# 추가 후 결과를 출력해 셋이 수정 가능한 자료구조임을 확인합니다.
print(fruits)

# 마지막으로 키와 값을 함께 관리하는 딕셔너리 섹션으로 넘어갑니다.
print("\n" + "=" * 50)
# 딕셔너리 제목을 출력합니다.
print("4. 딕셔너리")
# 역시 같은 형식의 구분선을 사용합니다.
print("=" * 50)

# 사람 한 명의 정보를 키와 값 형태로 정리한 딕셔너리를 만듭니다.
person = {
    "name": "철수",
    "age": 25,
    "height": 175.5,
}

# 키를 직접 지정하면 해당 키에 연결된 값을 바로 꺼낼 수 있습니다.
print(person["name"])
# get을 쓰면 없는 키를 찾을 때도 기본값을 돌려받아 에러를 피할 수 있습니다.
print(person.get("address", "정보없음"))

# 새 키를 넣으면 딕셔너리에 정보가 추가됩니다.
person["weight"] = 70
# 기존 키에 새 값을 넣으면 수정도 같은 방식으로 처리됩니다.
person["age"] = 26
# 추가와 수정이 반영된 전체 딕셔너리를 출력합니다.
print(person)

# keys는 딕셔너리의 키 목록을 보여 주는 뷰를 돌려줍니다.
print(list(person.keys()))
# values는 값들만 모아서 보여 줍니다.
print(list(person.values()))

# in 연산자로 특정 키가 존재하는지 빠르게 확인할 수 있습니다.
print("name 있나?", "name" in person)

# 딕셔너리 안에 또 딕셔너리를 넣으면 구조화된 데이터를 자연스럽게 표현할 수 있습니다.
students = {
    "철수": {"age": 25, "score": 85},
    "영희": {"age": 24, "score": 92},
}
# 중첩된 딕셔너리에서도 키를 차례대로 따라가면 원하는 값에 도달할 수 있습니다.
print("영희 점수:", students["영희"]["score"])

# 마무리 문장을 출력해 자료구조 네 종류 실습이 끝났음을 알려 줍니다.
print("\n실습 끝! 4가지 그릇을 자유자재로 다루실 수 있게 되시면 1장 절반 끝입니다.")
```
